int commonCharacterCount(char * s1, char * s2) {
    int common = 0;
    for(int i=0; i<strlen(s1);i++){
        for(int j=0; j<strlen(s2);j++){
            if (s1[i] == s2[j]){
                common++;
                s2[j] = ' ';
                break;
            }
        }
    }
    return common;
}
